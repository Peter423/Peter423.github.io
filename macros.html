Макрос - последовательность команд, запись которой автоматизированна и выполняется комбинацией клавиш
Макрос замена заменяет цвет всего текста на странице на голубой (Ctrl+ц)
Макрос втавка неогранич. вставляет в поле ссылку на урок по макросам (Ctrl+ч)
Макрос вставка огранич. вставляет текст только на первом листе (Ctrl+ы)
Макрос Каппа заменяет слово каппа на слово приди (Ctrl+у) Sub МакросЗамена()
'
' МакросЗамена Макрос
'
' Сочетание клавиш: Ctrl+ц
'
With Selection.Font
.ThemeColor = xlThemeColorAccent1
.TintAndShade = -0.249977111117893
End With
Cells.Select
Range("A2").Activate
With Selection.Font
.Color = -1003520
.TintAndShade = 0
End With
With Selection.Font
.Color = -1003520
.TintAndShade = 0
End With
End Sub
Sub МакросВставкаНеогранич()
'
' МакросВставкаНеогранич Макрос
'
' Сочетание клавиш: Ctrl+ч
'
ActiveCell.FormulaR1C1 = _
"https://www.youtube.com/watch?time_continue=6&v=g1l61o1CTwI"
Range("F17").Select
End Sub
Sub МакросВставкаОгранич()
'
' МакросВставкаОгранич Макрос
'
' Сочетание клавиш: Ctrl+ы
'
If Worksheets(ActiveSheet.Name).Name = "Лист1" Then
ActiveCell.FormulaR1C1 = _
"If Worksheets(ActiveSheet.Name).Name = ""Лист1"" Then (МАКРОС) End if"
End If
End Sub


Sub МакросКаппа()
'
' МакросКаппа Макрос
'
' Сочетание клавиш: Ctrl+у
'
Cells.Replace What:="каппа", Replacement:="приди", LookAt:=xlPart, _
SearchOrder:=xlByRows, MatchCase:=False, SearchFormat:=False, _
ReplaceFormat:=False
End Sub
